package Begin;


public class m {
	public static void main(String[]args) {//String 클래스 사용예제
		String proverb="A barking dog";
		String s1,s2,s3,s4;
		
		System.out.println("문자열 길이 ="+proverb.length());
		
		s1=proverb.concat("never Bites");//문자열 결합
		s2=proverb.replace("b","B");//문자교환
		s3=proverb.substring(2,5);//부분문자열 추출
		s4=proverb.toUpperCase();//대문자로 변환
		
		System.out.println(proverb);//원문
		System.out.println(s1);//concat적용
		System.out.println(s2);//replace적용
		System.out.println(s3);//substring적용
		System.out.println(s4);//toUpperCase적용
	}
}
